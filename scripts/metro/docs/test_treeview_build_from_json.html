            <div class="treeview" data-role="treeview" id="treeview1"></div>

            <script>
                var data = {
                    object1: "Leaf 1",
                    object2: {
                        subobject_2_1: "Leaf 2.1",
                        subobject_2_2: "Leaf 2.2",
                        subobject_2_3: "Leaf 2.2"
                    },
                    object3: {
                        subobject3_1: "Leaf 3.1",
                        subobject3_2: "Leaf 3.2",
                        subobject3_3: "Leaf 3.2"
                    }
                };

                $(function(){
                    var treeview = $("#treeview1").data("treeview");
                    var root, node;

                    root = treeview.addLeaf(false, "Root");

                    add(root, data);

                    function add(parent, data){
                        $.each(data, function(key, value){
                            if (typeof value === 'string') {
                                treeview.addLeaf(parent, this);
                            } else if (typeof value === 'object') {
                                node = treeview.addLeaf(parent, "Node " + key);
                                add(node, value);
                            }
                        });
                    }
                });
            </script>
